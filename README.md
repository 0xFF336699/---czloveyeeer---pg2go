# pg2go

go build 

./pg2go --help

-dbname string
数据名,默认dev_erp (default "dev_erp")

-host string
主机名,默认127.0.0.1 (default "127.0.0.1")

-out_dir string
生成文件目录 (default "gen/model")

-password string
密码,默认cap@2022 (default "cap@2022")

-port int
端口号,默认5432 (default 5432)

-sslmode string
模式,默认disable (default "disable")

-table string
表名，默认为空

-user string
用户名,默认postgres (default "postgres")
